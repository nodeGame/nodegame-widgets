Controls constructor

Control wraps a collection of user-input controls

Params
options object Optional. Configuration options which is stored and forwarded to Controls.init. The options object can have the following attributes:
  • Any option that can be passed to W.List constructor.
  • change: Event to fire when contents change.
  • features: Collection of collection attributes for individual controls.
  • submit: Description of the submit button. If submit.id is defined, the button will get that id and the text on the button will be the text in submit.name. If submit is a string, it will be the text on the button.
  • attributes: Attributes of the submit button.

    Controls.prototype.init = function(options) {
        this.hasChanged = false; // TODO: should this be inherited?
        if ('undefined' !== typeof options.change) {
            if (!options.change) {
                this.changeEvent = false;
            }
            else {
                this.changeEvent = options.change;
            }
        }
        this.list = new W.List(options);
        this.listRoot = this.list.getRoot();

        if (options.features) {
            this.features = options.features;
            this.populate();
        }
    };

            this.submit = W.add('button', this.bodyDiv,
                                J.merge(this.options.attributes, {
